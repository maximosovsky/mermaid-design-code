    %% ============================================================
    %% 1. WELCOME
    %% ============================================================
    START["🆕 User opens PhoneOut"]

    subgraph WELCOME_BLK["👋 Welcome"]
        direction LR
        WELCOME["Bot greeting
+ user name"]
        BUTTONS["🇫🇷 Louvre
🇺🇸 Rickroll"]
        TIP["2s → Tip:
type any number"]

        WELCOME --> BUTTONS --> TIP
    end

    START --> WELCOME_BLK

    %% ============================================================
    %% 2. THREE-WAY CHOICE
    %% ============================================================
    TIP --> CHOICE{"What does the user dial?"}

    CHOICE -->|"🇫🇷 +33140205317"| LOUVRE_FLOW
    CHOICE -->|"🇺🇸 +12484345508"| RICK_FLOW
    CHOICE -->|"Any other number"| CUSTOM_CALL

    %% ============================================================
    %% 3a. LOUVRE PATH (left column)
    %% ============================================================
    subgraph LOUVRE_BLK["🏛️ Louvre Path — 2 free calls"]
        direction TB
        LOUVRE_FLOW["☎️ Dial Louvre"]
        L_DC{"Redis: used < 2?"}
        L_YES["✅ free: true
Twilio WebRTC → 50s max"]
        L_INCR["INCR demo:+331..."]
        L_MSG["Bot summary:
☎️ +33140205317
🇫🇷 France · Nm Ns"]
        L_NO["❌ free: false"]

        LOUVRE_FLOW --> L_DC
        L_DC -->|"Yes"| L_YES
        L_YES --> L_INCR --> L_MSG
        L_DC -->|"No (used ≥ 2)"| L_NO
    end

    %% ============================================================
    %% 3b. RICKROLL PATH (right column)
    %% ============================================================
    subgraph RICK_BLK["🎵 Rickroll Path — 2 free calls"]
        direction TB
        RICK_FLOW["☎️ Dial Rickroll"]
        R_DC{"Redis: used < 2?"}
        R_YES["✅ free: true
Twilio WebRTC → 50s max"]
        R_INCR["INCR demo:+124..."]
        R_MSG["Bot summary:
☎️ +12484345508
🇺🇸 US · Nm Ns"]
        R_NO["❌ free: false"]

        RICK_FLOW --> R_DC
        R_DC -->|"Yes"| R_YES
        R_YES --> R_INCR --> R_MSG
        R_DC -->|"No (used ≥ 2)"| R_NO
    end

    %% ============================================================
    %% 3c. RICKROLL EASTER EGG
    %% ============================================================
    subgraph EASTER_BLK["🎸 Easter Egg (once per user)"]
        direction LR
        EASTER_CHECK{"rickroll_sent?"}
        EASTER_SEND["sendPhoto
Rick Astley + caption
Set flag = 1"]
        EASTER_SKIP["Skip"]

        EASTER_CHECK -->|"null"| EASTER_SEND
        EASTER_CHECK -->|"= 1"| EASTER_SKIP
    end

    R_MSG --> EASTER_CHECK

    %% ============================================================
    %% 4. PAYMENT WALL (converge point)
    %% ============================================================
    subgraph PAY_BLK["💳 Payment Wall — 300⭐"]
        direction TB
        PAY_GUARD["balance ≤ 0 AND !isDemo"]
        PAY_TIMER["5s countdown"]
        PAY_END["endCall()"]
        PAY_INVOICE["openStarsPayment(300)
💳 Telegram Stars
300⭐ = 30 min US"]

        PAY_GUARD --> PAY_TIMER --> PAY_END --> PAY_INVOICE
    end

    %% Non-demo number goes straight to payment
    CUSTOM_CALL["User dials non-demo number
isDemo = false, balance = 0"] --> PAY_GUARD

    %% Demo exhausted goes to payment
    L_NO --> PAY_GUARD
    R_NO --> PAY_GUARD

    %% After free call, user chooses again
    L_MSG -. "next call" .-> CHOICE
    EASTER_SEND -. "next call" .-> CHOICE
    EASTER_SKIP -. "next call" .-> CHOICE

    %% ============================================================
    %% SERVICES (external, bottom)
    %% ============================================================
    TWILIO["📞 Twilio WebRTC"]
    REDIS["🗄️ Upstash Redis"]
    TG_BOT["🤖 Telegram Bot API"]

    L_YES -. "call" .-> TWILIO
    R_YES -. "call" .-> TWILIO
    L_DC -. "GET/INCR" .-> REDIS
    R_DC -. "GET/INCR" .-> REDIS
    EASTER_CHECK -. "GET/SET" .-> REDIS
    L_MSG -. "sendMessage" .-> TG_BOT
    R_MSG -. "sendMessage" .-> TG_BOT
    EASTER_SEND -. "sendPhoto" .-> TG_BOT
